# Manual inputting of the data 
# Since dimension of pdf_teams is 288 while pdf_teamname is 299, missing data for 11 teams so cannot sort both and concatenate 
# For each value in the pdf_teams data value, assuming that they are unique, they will traverse the pdf_teamname until
# They hit the target row and the for loop will break and start with the next row of the pdf_teams

for index,row in pdf_teams.iterrows():
    team1 = index
    for index1,row1 in pdf_teamname.iterrows():
        team2 = row1['team_id']
        if team1 == team2:
            pdf_teamname.loc[pdf_teamname.team_id == team1,'buildUpPlaySpeed'] = row['buildUpPlaySpeed']
            pdf_teamname.loc[pdf_teamname.team_id == team1,'buildUpPlayDribbling'] = row['buildUpPlayDribbling']
            pdf_teamname.loc[pdf_teamname.team_id == team1,'buildUpPlayPassing'] = row['buildUpPlayPassing']
            pdf_teamname.loc[pdf_teamname.team_id == team1,'chanceCreationPassing'] = row['chanceCreationPassing']
            pdf_teamname.loc[pdf_teamname.team_id == team1,'chanceCreationCrossing'] = row['chanceCreationCrossing']
            pdf_teamname.loc[pdf_teamname.team_id == team1,'chanceCreationShooting'] = row['chanceCreationShooting']
            pdf_teamname.loc[pdf_teamname.team_id == team1,'defencePressure'] = row['defencePressure']
            pdf_teamname.loc[pdf_teamname.team_id == team1,'defenceAggression'] = row['defenceAggression']
            pdf_teamname.loc[pdf_teamname.team_id == team1,'defenceTeamWidth'] = row['defenceTeamWidth']
            break
        else:
            continue
